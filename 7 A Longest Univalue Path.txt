class TreeNode { 
    int val; 
    TreeNode left, right; 
    TreeNode(int x) { val = x; } 
} 
public class LongestUnivaluePath { 
    static int maxLen = 0; 
    public static int longestUnivaluePath(TreeNode root) { 
        dfs(root); 
        return maxLen; 
    } 
    private static int dfs(TreeNode node) { 
        if (node == null) return 0; 
        int left = dfs(node.left); 
        int right = dfs(node.right); 
        int leftPath = 0, rightPath = 0; 
        if (node.left != null && node.left.val == node.val) leftPath = left + 1; 
        if (node.right != null && node.right.val == node.val) rightPath = right + 1; 
        maxLen = Math.max(maxLen, leftPath + rightPath); 
        return Math.max(leftPath, rightPath); 
    } 
    public static void main(String[] args) { 
        TreeNode root = new TreeNode(5); 
        root.left = new TreeNode(4); 
        root.right = new TreeNode(5); 
        root.left.left = new TreeNode(4); 
        root.left.right = new TreeNode(4); 
        root.right.right = new TreeNode(5); 
        System.out.println("Longest Univalue Path Length: " + longestUnivaluePath(root)); 
    } 
}
